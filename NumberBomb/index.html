#include<bits/stdc++.h>
#include<windows.h>
#include<conio.h>
#include<string>
using namespace std;
void rl2()//规则2
{
    cout<<"2.你需要在这个范围内打出你想到的整数(最大值与最小值不算)"<<endl;
}
void st()//开始界面
{
    cout<<"**欢迎来到数字炸弹**"<<endl
        <<"以下是本游戏的规则:"<<endl
        <<"1.你需要输入一个范围"<<endl;
    rl2();
    cout<<"3.如果数字和系统给的随机数相同就算失败"<<endl
        <<"4.建议多人游玩哦"<<endl
        <<endl
        <<"请按任意键开始游戏...";
}
void cl()//清屏
{
    system("cls");
}
void ps()//暂停
{
    system("pause>nul");
}
void sl(int number)//等待
{
    Sleep(number);
}
int main()//主程序
{
    srand(time(0));//撒随机种子
    long long bomb=0,maximum=0,minimum=0,input=0;
    string cmd="0",scmd="0";
    bool newg=0;
    st();
    cout<<" (3)";
    sl(1000);
    cl();
    st();
    cout<<" (2)";
    sl(1000);
    cl();
    st();
    cout<<" (1)";
    sl(1000);
    cl();
    st();
    ps();
    cl();
    while(true)//开始无限循环
    {
        cout<<"请输入最小值:";
        cin>>minimum;//输入最小值
        while(maximum<=minimum+1)
        {
            cout<<"请输入最大值:";
            cin>>maximum;//输入最大值
        }
        while(bomb<=minimum)
        {
            bomb=rand()%maximum;//设置炸弹
        }
        while(true)//一局游戏
        {
            if(newg==1)
            {
                break;
            }
            cout<<"请输入在"<<minimum<<"-"<<maximum<<"之内的数字:";
            cin>>input;
            if(input>maximum||input<minimum)//数字在范围之外
            {
                rl2();
                cout<<"        ~~~~~~~~~~~~~~~~~~~~~~~~~~~~                    "<<endl
                    <<endl
                    <<"请按任意键返回游戏...";
                ps();
                cl();
                continue;
            }
            if(input==maximum||input==minimum)//输入了最小值或最大值
            {
                rl2();
                cout<<"                                     ~~~~~~~~~~~~~~~~~~ "<<endl
                    <<endl
                    <<"请按任意键返回游戏...";
                ps();
                cl();
                continue;
            }
            if(input>bomb)//大于炸弹
            {
                maximum=input;
            }
            if(input<bomb)//小于炸弹
            {
                minimum=input;
            }
            if(input==bomb)//等于炸弹
            {
                cout<<"你踩到了炸弹!"<<endl;
                cl();
                ps();
                break;
            }
            cl();
        }
    }
    return 0;
}

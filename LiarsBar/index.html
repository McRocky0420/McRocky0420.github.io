int checkalive()
{
    int alive=0;
    for(int i=1; i<=player; i++)
    {
        if(life[i]=="活着")
        {
            alive++;
        }
    }
    return alive;
}
void fpai()
{
    for(int fplayer=1; fplayer<=player; fplayer++)
    {
        if(life[fplayer]=="活着")
        {
            for(int fnum=1; fnum<=5; fnum++)
            {
                ok=false;
                while(ok==false)
                {
                    num=rand()%9;
                    switch(num)
                    {
                        case 0:
                            if(q>0)
                            {
                                q--;
                                pai[fplayer][fnum]='q';
                                ok=true;
                            }
                            break;
                        case 1:
                            if(k>0)
                            {
                                k--;
                                pai[fplayer][fnum]='k';

                            }
                            break;
                        case 2:

                            }

                            {

                                ok=true;
                            }
                            break;
                        case 4:
                            if(W>0)
                            {
                                W--;

                                ok=true;
                            }
                            break;
                        case 5:
                            if(w>0)
                            {
                                w--;
                                pai[fplayer][fnum]='w';
                                ok=true;
                            }
                            break;
                        case 6:
                            if(Q>0)

                                Q--;
                                if(reallypai=='q')
                                {
                                    pai[fplayer][fnum]='Q';
                                }
                                else
                                {
                                    pai[fplayer][fnum]='q';
                                }
                                ok=true;
                            }
                            break;
                        case 7:
                            if(K>0)
                            {
                                K--;
                                if(reallypai=='k')
                                {
                                    pai[fplayer][fnum]='K';
                                }
                                else
                                {
                                    pai[fplayer][fnum]='k';
                                }
                                ok=true;
                            }
                            break;
                        case 8:
                            if(A>0)
                            {
                                A--;
                                if(reallypai=='a')
                                {
                                    pai[fplayer][fnum]='A';
                                }
                                else
                                {
                                    pai[fplayer][fnum]='a';
                                }
                                ok=true;
                            }
                            break;
                    }
                }
            }
        }
    }
    cout<<"所有玩家的数据:"<<endl;
    for(int file=1; file<=player; file++)
    {
        cout<<"玩家"<<file<<":"<<guntime[file]<<"/6 "<<life[file]<<endl;
    }
    cout<<"|?|    |"<<reallypai<<"|"<<endl
        <<endl;
    for(int i=1; i<=player; i++)
    {
        cout<<"("<<i<<") ";
        if(life[i]=="活着")
        {
            for(int b=1; b<=5; b++)
            {
                Sleep(150);
                cout<<"|?|";
            }
        }
        else
        {
            cout<<"该玩家已死亡,不发牌";
        }
        cout<<endl
            <<endl;
    }
}

    A=1;

    lifepeople=player;

    quit=false;
    for(int i=1; i<=player; i++)
    {
        guntime[i]=0;

void kgun(int gunplayer)
{
    guntime[gunplayer]++;
    doplayer=0;
    cout<<"玩家"<<gunplayer<<"正在自枪";
    for(int i=1; i<=3; i++)
    {
        cout<<".";
        Sleep(500);
    }
    switch(guntime[gunplayer])
    {
        case 0:
            gunrand=rand()%6;
            if(gunrand==0)
            {
                life[gunplayer]="死了";
            }
            break;
        case 1:
            gunrand=rand()%5;
            if(gunrand==0)
            {
                life[gunplayer]="死了";
            }
            break;
        case 2:
            gunrand=rand()%4;
            if(gunrand==0)
            {
                life[gunplayer]="死了";
            }
            break;
        case 3:
            gunrand=rand()%3;
            if(gunrand==0)
            {
                life[gunplayer]="死了";
            }
            break;
        case 4:
            gunrand=rand()%2;
            if(gunrand==0)
            {
                life[gunplayer]="死了";
            }
            break;
        case 5:
            gunrand=rand()%1;
            if(gunrand==0)
            {
                life[gunplayer]="死了";
            }
            break;
    }
    quit=true;
    cout<<"是"<<(life[gunplayer]=="死了"?"实弹!":"空弹!")<<endl
        <<"请按任意键下一局..."<<endl;
    system("pause>nul&&cls");
}
void playerdo()
{
    if(checkalive()==1)
    {
        system("cls");
        cout<<"========================================"<<endl
            <<"                游戏结束!               "<<endl
            <<"========================================"<<endl
            <<"    场上玩家数:1                        "<<endl;
        for(int i=1; i<=player; i++)
        {
            if(life[i]=="活着")
            {
                cout<<"    玩家名称:玩家"<<i<<endl;
                break;
            }
        }
        cout<<"========================================"<<endl
            <<"请按任意键退出游戏..."<<endl;
        system("pause>nul");
        return;
    }
    cout<<"当前真牌:|"<<reallypai<<"|"<<endl
        <<"出牌堆:"<<endl;
    for(int i=1; i<=5; i++)
    {
        if(paiout[lplayer][i]==true)
        {
            cout<<"|?| ";
        }
    }
    cout<<endl
        <<"接下来将是玩家"<<doplayer<<"的回合,开始前请确认其他玩家已经死去..."<<endl;
    system("pause>nul&&cls");
    cout<<"玩家"<<doplayer<<"回合"<<endl
        <<"当前真牌:|"<<reallypai<<"|"<<endl
        <<"所有玩家的数据:"<<endl;
    for(int file=1; file<=player; file++)
    {
        cout<<"玩家"<<file<<":"<<guntime[file]<<"/6 "<<life[file]<<(file==doplayer?" (当前玩家)":"")<<endl;
    }
    cout<<"所有玩家的牌:"<<endl
        <<"编号 1   2   3   4   5"<<endl;
    for(int splayer=1; splayer<=player; splayer++)
    {
        cout<<"("<<splayer<<") ";
        if(life[splayer]=="活着")
        {
            for(int snum=1; snum<=5; snum++)
            {
                if(paiout[splayer][snum]==false)
                {
                    cout<<"|";
                    if(splayer==doplayer)
                    {
                        cout<<pai[splayer][snum];
                    }
                    else
                    {
                        cout<<"?";
                    }
                    cout<<"| ";
                }
                else
                {
                    cout<<"    ";
                }
            }
        }
        else
        {
            cout<<"该玩家已死亡,无牌";
        }
        cout<<(splayer==doplayer?" (当前玩家)":"")<<endl
            <<endl;
    }
    cout<<"出牌堆:"<<endl;
    for(int i=1; i<=5; i++)
    {
        if(paiout[lplayer][i]==true)
        {
            cout<<"|?| ";
        }
    }
    cout<<endl
        <<"A:出牌"<<endl
        <<(times!=1?"B:质疑(Liar)":"这是本局第一次操作，无法质疑!")<<endl
        <<"C:放弃这一局(自枪)"<<endl
        <<"请键入你要进行的操作:";
    while(!_kbhit())
    {
    }
    switch(_getch())
    {
        case 'a':
        case 'A':
        {
            cout<<endl;
            outnum=painum[doplayer]+1;
            while((outnum>painum[doplayer])||(outnum>3)||(outnum<=0))
            {
                cout<<"请输入你要出牌的数量(1-"<<(painum[doplayer]>3?3:painum[doplayer])<<"):";
                cin>>outnum;
            }
            for(int outsum=1; outsum<=outnum; outsum++)
            {
                outpainum=0;
                while(outpainum<1||outpainum>5||paiout[doplayer][outpainum])
                {
                    for(int i=1; i<=5; i++)
                    {
                        if(!paiout[doplayer][i])
                        {
                            cout<<"["<<i<<"]:"<<pai[doplayer][i]<<" ";
                        }
                    }
                    cout<<"请输入你要出的第"<<outsum<<"个牌的位置(1-5):";
                    cin>>outpainum;
                    outpai=pai[doplayer][outpainum];
                    if(outpainum<1||outpainum>5)
                    {
                        cout<<"无效位置!请输入1-5..."<<endl;
                    }
                    else if(outnum!=1&&((pai[doplayer][outpainum]=='Q')||(pai[doplayer][outpainum]=='K')||(pai[doplayer][outpainum]=='A')))
                    {
                        cout<<"魔鬼牌只能单出!"<<endl
                            <<"是否只出这张魔鬼牌(Y/N)?";
                        while(!_kbhit())
                        {
                        }
                        switch(_getch())
                        {
                            case 'y':
                            case 'Y':
                                outnum=1;
                                outpais[1]=pai[doplayer][outpainum];
                                break;
                            case 'n':
                            case 'N':
                                outsum--;
                                break;
                        }
                    }
                    else if(paiout[doplayer][outpainum])
                    {
                        cout<<"这张牌已经出过了!"<<endl;
                    }
                }
                paiout[doplayer][outpainum]=true;
                outpais[outsum]=outpai;
                cout<<"打出牌:"<<outpais[outsum]<<endl;
            }
            painum[doplayer]-=outnum;
            cout<<"请按任意键结束..."<<endl;
            system("pause>nul&&cls");
            break;
        }
        case 'b':
        case 'B':
        {
            if(times==1)
            {
                cout<<"无效操作!"<<endl;
                system("cls");
                cout<<"等待所有玩家回来..."<<endl;
                system("pause>nul&&cls");
                cout<<"因为玩家1试图质疑空玩家的牌,疑似规避差牌被质疑而自枪,所以作为惩罚,ta必须自枪"<<endl
                    <<"请按任意键让ta自枪..."<<endl;
                system("pause>nul&&cls");
                kgun(1);
                break;
            }
            system("cls");
            cout<<"等待所有玩家回来..."<<endl;
            system("pause>nul&&cls");
            cout<<"玩家"<<doplayer<<"决定质疑上一位玩家的牌"<<endl
                <<"请按任意键开牌"<<endl;
            system("pause>nul&&cls");
            for(int i=1; i<=outnum; i++)
            {
                cout<<"|?| ";
            }
            Sleep(500);
            system("cls");
            for(int i=1; i<=outnum; i++)
            {
                cout<<" |  ";
            }
            Sleep(500);
            system("cls");
            for(int i=1; i<=outnum; i++)
            {
                cout<<"|"<<outpais[i]<<"| ";
            }
            cout<<"请按任意键验牌..."<<endl;
            system("pause>nul&&cls");
            noliar=true;
            mogui=false;
            for(int i=1; i<=outnum; i++)
            {
                cout<<"|"<<outpais[i]<<"| ";
            }
            cout<<endl;
            for(int i=1; i<=outnum; i++)
            {
                if((outpais[i]=='Q')||(outpais[i]=='K')||(outpais[i]=='A'))
                {
                    mogui=true;
                    cout<<">v< ";
                }
                else if((outpais[i]=='W')||(outpais[i]=='w')||(outpais[i]=='j')||(outpais[i]==reallypai))
                {
                    cout<<" v  ";
                }
                else
                {
                    noliar=false;
                    cout<<" x  ";
                }
                Sleep(500);
            }
            cout<<endl;
            if(mogui==true)
            {
                cout<<"玩家"<<lplayer<<"出了魔鬼牌!"<<endl
                    <<"请按任意键让除玩家"<<lplayer<<"之外的所有玩家自枪..."<<endl;
                system("pause>nul&&cls");
                for(int i=1; i<=player; i++)
                {
                    if(i!=lplayer)
                    {
                        kgun(i);
                    }
                }
            }
            else if(noliar==true)
            {
                cout<<"玩家"<<lplayer<<"是好人!"<<endl
                    <<"请按任意键让玩家"<<doplayer<<"自枪..."<<endl;
                system("pause>nul&&cls");
                kgun(doplayer);
            }
            else
            {
                cout<<"玩家"<<lplayer<<"是Liar!"<<endl
                    <<"请按任意键让玩家"<<lplayer<<"自枪..."<<endl;
                system("pause>nul&&cls");
                kgun(lplayer);
            }
            break;
        }
        case 'c':
        case 'C':
        {
            system("cls");
            cout<<"等待所有玩家回来..."<<endl;
            system("pause>nul&&cls");
            cout<<"玩家"<<doplayer<<"决定放弃这一局,所以ta要自枪!"<<endl
                <<"请按任意键让ta自枪"<<endl;
            system("pause>nul&&cls");
            kgun(doplayer);
            break;
        }
        default:
        {
            system("cls");
            cout<<"等待所有玩家回来..."<<endl;
            system("pause>nul&&cls");
            cout<<"玩家"<<doplayer<<"按下了非法按键,疑似规避差牌被质疑而自枪,所以作为惩罚,ta必须自枪!"<<endl
                <<"请按任意键让ta自枪"<<endl;
            system("pause>nul&&cls");
            kgun(doplayer);
            break;
        }
    }
}
int main()
{
    system("mode con:cols=40 lines=25");
    srand(time(0));
    system("cls");
    cout<<"========================================"<<endl
        <<"            骗子酒馆(简易版)"<<endl
        <<"========================================"<<endl
        <<"1.每轮游戏会出一张真牌,你会有一堆牌,尽量保证不被他人看见"<<endl
        <<"2.小写字母表示普通牌,大写字母表示魔鬼牌,W表示大王,w表示小王,大小王和j也是真牌"<<endl
        <<"3.如果魔鬼牌被质疑,除出牌者之外的所有人都要自枪,恶魔牌只能单出"<<endl
        <<"4.每次你可以选择任意不多于3张牌打出,同时你需要大喊真牌的点数及出牌的个数"<<endl
        <<"5.如果你不相信上家的出牌,可以大喊'liar'来质疑"<<endl
        <<"6.如果质疑的牌都是真牌,则质疑者自枪,否则被质疑者自枪"<<endl
        <<"7.枪有1个实弹和5个空弹,若为空弹,则安然无事,若为实弹,则死亡"<<endl
        <<"8.轮到任意玩家回合时,其他玩家请死去"<<endl
        <<"========================================"<<endl
        <<"*请按任意键开始游戏...                  "<<endl;
    system("pause>nul&&cls");
    while(player<2||player>4)
    {
        cout<<"有几个人玩(1<player<5)?"<<endl;
        cin>>player;
    }
    csh();
    while(1)
    {
        csh2();
        cout<<"请按任意键开真牌..."<<endl;
        system("pause>nul&&cls");
        kreallypai();
        cout<<"所有玩家的数据:"<<endl;
        for(int file=1; file<=player; file++)
        {
            cout<<"玩家"<<file<<":"<<guntime[file]<<"/6 "<<life[file]<<endl;
        }
        cout<<"真牌是|"<<reallypai<<"|"<<endl
            <<"请按任意键发牌..."<<endl;
        system("pause>nul&&cls");
        fpai();
        cout<<"请按任意键开始出牌"<<endl;
        system("pause>nul&&cls");
        while(!quit)
        {
            times++;
            doplayer++;
            if(doplayer>player)
            {
                doplayer=1;
            }
            lplayer=doplayer-1;
            if(doplayer==1)
            {
                lplayer=player;
            }
            if(painum[doplayer]==0||life[doplayer]=="死了")
            {
                continue;
            }
            playerdo();
            system("cls");
            if(checkalive()==1)
            {
                return 0;
            }
        }
    }
    return 0;
}
